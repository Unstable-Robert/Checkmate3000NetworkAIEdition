001package edu.mccc.cos210.fp2014.cm.piece;
002
003import java.util.ArrayList;
004import javax.xml.bind.annotation.XmlElement;
005import javax.xml.bind.annotation.XmlRootElement;
006
007import edu.mccc.cos210.fp2014.cm.game.Board;
008import edu.mccc.cos210.fp2014.cm.util.Color;
009
010/**
011 * An abstract class representing an individual piece on the board.
012 */
013@XmlRootElement
014public abstract class Piece implements Cloneable {
015        @XmlElement
016        protected int xLoc;
017        @XmlElement
018        protected int yLoc;
019        @XmlElement
020        protected Color color;
021        @XmlElement
022        protected int uID;
023        @XmlElement
024        protected boolean selected;
025        /**
026         * Constructor to be implemented by subclasses which gets a list of possible moves.
027         */
028        public Piece(){
029        }
030        public Piece(int x, int y, Color c, int iD){
031        }
032        public Piece(int x, int y, Color c, int iD, boolean s){
033        }
034        public abstract ArrayList<PossibleTile> getPossibleTiles(Board b);
035        /**
036         * Changes the location of the piece.
037         */
038        public void setLocation(int x, int y) {
039        }
040        public int getX(){
041                return this.xLoc;
042        }
043        public int getY(){
044                return this.yLoc;
045        }
046        public Color getColor(){
047                return this.color;
048        }
049        public int getUID(){
050                return this.uID;
051        }
052        public boolean isSelected(){
053                return this.selected;
054        }
055        /**
056         * Changes whether or not the piece is selected.
057         */
058        public void setSelected(boolean b) {
059        }
060        public abstract Piece clone();
061}
