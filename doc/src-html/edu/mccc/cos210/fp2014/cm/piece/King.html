003import java.util.ArrayList;
004import javax.xml.bind.annotation.XmlElement;
005import javax.xml.bind.annotation.XmlRootElement;
006
007import edu.mccc.cos210.fp2014.cm.game.Board;
008import edu.mccc.cos210.fp2014.cm.util.Color;
009
010/**
011 * Represents the king piece and its movements.
012 */
013@XmlRootElement
014public class King extends Piece {
015        @XmlElement
016        private boolean canCastle;
017        public King(int x, int y, Color c, int iD){
018                super(x,y,c,iD);
019                this.canCastle = true;
020        }
021        public King(int x, int y, Color c, int iD, boolean s){
022                super(x,y,c,iD,s);
023                this.canCastle = true;
024        }
025        public King(int x, int y, Color c, int iD, boolean s, boolean castle){
026                super(x,y,c,iD,s);
027                this.canCastle = castle;
028        }
029        /**
030         * Gets possible tiles that this piece can move on the given board.
031         * @param board the board that is checked for possible moves
032         * @return all of the possible tiles to which this piece can move.
033         */
034        @Override
035        public ArrayList<PossibleTile> getPossibleTiles(Board board) {
036                return null;
037        }
038        /**
039         * Sees if the king can castle to the left.
040         * Returns true if the king has not moved, there are no pieces in the way, 
041         * and if the king is in check, moving through check, or into check.
042         */
043        public boolean canCastleLeft(){
044                return false;
045        }
046        /**
047         * Sees if the king can castle to the right.
048         * Returns true if the king has not moved, there are no pieces in the way, 
049         * and if the king is in check, moving through check, or into check.
050         */
051        public boolean canCastleRight(){
052                return false;
053        }
054        public King clone(){
055                return new King(this.getX(), 
056                                this.getY(), 
057                                this.getColor(), 
058                                this.getUID(), 
059                                this.isSelected(), 
060                                this.canCastle());
061        }
062}
