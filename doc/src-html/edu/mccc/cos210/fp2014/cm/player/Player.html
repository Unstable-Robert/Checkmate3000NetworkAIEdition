001package edu.mccc.cos210.fp2014.cm.player;
002
003import java.util.Observer;
004
005import edu.mccc.cos210.fp2014.cm.game.GameModel;
006import edu.mccc.cos210.fp2014.cm.piece.Piece;
007
008/**
009 * Abstract player class.
010 * Generic player object, containing the update model and update methods in addition to 
011 * some a game model and a color.
012 */
013public abstract class Player implements Observer{
014        protected GameModel gm;
015        protected boolean isWhite;
016        /**
017         * This class will update the game model.
018         */
019        protected abstract void updateModel(Piece p);
020}
